  213   /* Function: rtOneStep ========================================================
  214    *
  215    * Abstract:
  216    *      Perform one step of the model. This function is modeled such that
  217    *      it could be called from an interrupt service routine (ISR) with minor
  218    *      modifications.
  219    *
  220    *      This routine is modeled for use in a multitasking environment and
  221    *      therefore needs to be fully re-entrant when it is called from an
  222    *      interrupt service routine.
  223    *
  224    * Note:
  225    *      Error checking is provided which will only be used if this routine
  226    *      is attached to an interrupt.
  227    *
  228    */
  229   static int_T eventFlags[NUMST];
  230   
  231   #ifdef TWO_STEPS
  232   
  233   static void rt_OneStep1(RT_MODEL *S)
  234   #else
  235     static void rt_OneStep(RT_MODEL *S)
  236   #endif
  237   {
  238     int_T i;
  239     real_T tnext;
  240     int_T *sampleHit = rtmGetSampleHitPtr(S);
  241   
  242     /***********************************************
  243      * Check and see if base step time is too fast *
  244      ***********************************************/
  245     if (GBLbuf.isrOverrun++) {
  246       GBLbuf.stopExecutionFlag = 1;
  247       return;
  248     }
  249   
  250     /***********************************************
  251      * Check and see if error status has been set  *
  252      ***********************************************/
  253     if (rtmGetErrorStatus(S) != NULL) {
  254       GBLbuf.stopExecutionFlag = 1;
  255       return;
  256     }
  257   
  258     /* enable interrupts here */
  259   
  260     /************************************************************************
  261      * Update discrete events and buffer event flags locally so that ISR is *
  262      * re-entrant.                                                          *
  263      ************************************************************************/
  264     tnext = rt_SimUpdateDiscreteEvents(rtmGetNumSampleTimes(S),
  265       rtmGetTimingData(S),
  266       rtmGetSampleHitPtr(S),
  267       rtmGetPerTaskSampleHitsPtr(S));
  268     rtsiSetSolverStopTime(rtmGetRTWSolverInfo(S),tnext);
  269     for (i=FIRST_TID+1; i < NUMST; i++) {
  270       eventFlags[i] = sampleHit[i];
  271     }
  272   
  273     /*******************************************
  274      * Step the model for the base sample time *
  275      *******************************************/
  276     MdlOutputs(FIRST_TID);
  277     rtExtModeUploadCheckTrigger(rtmGetNumSampleTimes(S));
  278     rtExtModeUpload(FIRST_TID,rtmGetTaskTime(S, FIRST_TID));
  279   
  280   #ifdef TWO_STEPS
  281   
  282   }

  284   static void rt_OneStep2(RT_MODEL *S)
  285   {
  286     int_T i;
  287   
  288   #endif
  289   
  290     MdlUpdate(FIRST_TID);
  291     if (rtmGetSampleTime(S,0) == CONTINUOUS_SAMPLE_TIME) {
  292       rt_UpdateContinuousStates(S);
  293     } else {
  294       rt_SimUpdateDiscreteTaskTime(rtmGetTPtr(S),
  295         rtmGetTimingData(S), 0);
  296     }
  297   
  298   #if FIRST_TID == 1
  299   
  300     rt_SimUpdateDiscreteTaskTime(rtmGetTPtr(S),
  301       rtmGetTimingData(S),1);
  302   
  303   #endif
  304   
  305     /************************************************************************
  306      * Model step complete for base sample time, now it is okay to          *
  307      * re-interrupt this ISR.                                               *
  308      ************************************************************************/
  309     GBLbuf.isrOverrun--;
  310   
  311     /*********************************************
  312      * Step the model for any other sample times *
  313      *********************************************/
  314     for (i=FIRST_TID+1; i<NUMST; i++) {
  315       if (eventFlags[i]) {
  316         if (GBLbuf.overrunFlags[i]++) {  /* Are we sampling too fast for */
  317           GBLbuf.stopExecutionFlag= 1;   /*   sample time "i"?           */
  318           return;
  319         }
  320   
  321         MdlOutputs(i);
  322         rtExtModeUpload(i, rtmGetTaskTime(S,i));
  323         MdlUpdate(i);
  324         rt_SimUpdateDiscreteTaskTime(rtmGetTPtr(S),
  325           rtmGetTimingData(S),i);
  326   
  327         /* Indicate task complete for sample time "i" */
  328         GBLbuf.overrunFlags[i]--;
  329       }
  330     }
  331   
  332     rtExtModeCheckEndTrigger();
  333   }                                      /* end rtOneStep */

  337   /* Function: checkForErrors ===================================================
  338    *
  339    * Abstract:
  340    *      Check for errors and produce appropriate output if applicable.
  341    *
  342    */
  343   static int_T checkForErrors(RT_MODEL *S)
  344   {
  345     if (GBLbuf.errmsg) {
  346       asdWriteUserError("LCRT RTI: %s\n",GBLbuf.errmsg);
  347       errorFlag = 1;
  348       return 1;
  349     }
  350   
  351     if (GBLbuf.isrOverrun) {
  352       asdWriteUserError("LCRT RTI: %s: ISR overrun - base sampling rate is "
  353                         "too fast\n", QUOTE(MODEL));
  354       errorFlag = 1;
  355       return 1;
  356     }
  357   
  358     if (rtmGetErrorStatus(S) != NULL) {
  359       asdWriteUserError("LCRT RTI: %s\n", rtmGetErrorStatus(S));
  360       errorFlag = 1;
  361       return 1;
  362     }
  363   
  364   #ifdef MULTITASKING
  365   
  366     else {
  367       int_T i;
  368       for (i=1; i<NUMST; i++) {
  369         if (GBLbuf.overrunFlags[i]) {
  370           asdWriteUserError("LCRT RTI: %s ISR overrun - sampling rate is "
  371                             "too fast for sample time index %d\n",
  372                             QUOTE(MODEL), i);
  373           errorFlag = 1;
  374           return 1;
  375         }
  376       }
  377     }
  378   #endif
  379   
  380     return 0;
  381   }                                      /* end checkForErrors */
