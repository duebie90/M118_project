   76   /* Function: rt_SimInitTimingEngine ============================================
   77    * Abstract:
   78    *      This function is for use with single tasking or multitasking
   79    *      real-time systems.
   80    *
   81    *      Initializes the timing engine for a fixed-step real-time system.
   82    *      It is assumed that start time is 0.0.
   83    *
   84    * Returns:
   85    *      NULL     - success
   86    *      non-NULL - error string
   87    */
   88   const char *rt_SimInitTimingEngine(int_T rtmNumSampTimes,
   89     real_T rtmStepSize,
   90     real_T *rtmSampleTimePtr,
   91     real_T *rtmOffsetTimePtr,
   92     int_T *rtmSampleHitPtr,
   93     int_T *rtmSampleTimeTaskIDPtr,
   94     real_T rtmTStart,
   95     SimTimeStep *rtmSimTimeStepPtr,
   96     void **rtmTimingDataPtr)
   97   {
   98   
   99   #ifdef USE_RTMODEL
  100   
  101     /* In the USE_RTMODEL case this function does nothing */
  102     UNUSED_PARAMETER(rtmNumSampTimes);
  103     UNUSED_PARAMETER(rtmStepSize);
  104     UNUSED_PARAMETER(rtmSampleTimePtr);
  105     UNUSED_PARAMETER(rtmOffsetTimePtr);
  106     UNUSED_PARAMETER(rtmSampleHitPtr);
  107     UNUSED_PARAMETER(rtmSampleTimeTaskIDPtr);
  108     UNUSED_PARAMETER(rtmSimTimeStepPtr);
  109     UNUSED_PARAMETER(rtmTimingDataPtr);
  110     if (rtmTStart != 0.0) {
  111       return("Start time must be zero for real-time systems.  For non-zero start times you must use the Simulink solver module");
  112     } else {
  113       return(NULL);
  114     }
  115   
  116   #else                                  /* must be !USE_RTMODEL */
  117   
  118     int_T i;
  119     int *tsMap = rtmSampleTimeTaskIDPtr;
  120     real_T *period = rtmSampleTimePtr;
  121     real_T *offset = rtmOffsetTimePtr;
  122     int_T *sampleHit = rtmSampleHitPtr;
  123     real_T stepSize = rtmStepSize;
  124   
  125   #ifdef RT_MALLOC
  126   
  127     int_T numst = rtmNumSampTimes;
  128   
  129     /* In the dynamically allocated case, we allocate the data here */
  130     static const char_T *malloc_error;
  131     TimingData *td;
  132     malloc_error = "Memory allocation error";
  133     td = (TimingData *) malloc(sizeof(TimingData));
  134     if (!td) {
  135       return(malloc_error);
  136     }
  137   
  138     td->period = (real_T *) malloc(numst * sizeof(real_T));
  139     if (!td->period) {
  140       return(malloc_error);
  141     }
  142   
  143     td->offset = (real_T *) malloc(numst * sizeof(real_T));
  144     if (!td->offset) {
  145       return(malloc_error);
  146     }
  147   
  148     td->clockTick = (real_T *) malloc(numst * sizeof(real_T));
  149     if (!td->clockTick) {
  150       return(malloc_error);
  151     }
  152   
  153     td->taskTick = (int_T *) malloc(numst * sizeof(int_T));
  154     if (!td->taskTick) {
  155       return(malloc_error);
  156     }
  157   
  158     td->nTaskTicks = (int_T *) malloc(numst * sizeof(int_T));
  159     if (!td->nTaskTicks) {
  160       return(malloc_error);
  161     }
  162   
  163     if (rtmTStart != 0.0) {
  164       return("Start time must be zero for real-time systems.  For non-zero start times you must use the Simulink solver module");
  165     }
  166   
  167   #else                                  /* must be !RT_MALLOC */
  168   
  169     /* In the statically allocated case, we point to the static structure */
  170     TimingData *td;
  171     td = &td_struct;
  172   
  173     /* Also, we use the constant NUMST instead of what was passed in */
  174     rtmNumSampTimes = NUMST;
  175   
  176   #endif                                 /* !RT_MALLOC */
  177   
  178     if (rtmTStart != 0.0) {
  179       return("Start time must be zero for real-time systems.  For non-zero start times you must use the Simulink solver module");
  180     }
  181   
  182     *rtmSimTimeStepPtr = MAJOR_TIME_STEP;
  183     *rtmTimingDataPtr = (void*)&td;
  184     for (i = 0; i < rtmNumSampTimes; i++) {
  185       tsMap[i] = i;
  186       td->period[i] = period[i];
  187       td->offset[i] = offset[i];
  188       td->nTaskTicks[i] = (int_T)floor(period[i]/stepSize + 0.5);
  189       if (td->period[i] == CONTINUOUS_SAMPLE_TIME ||
  190           td->offset[i] == 0.0) {
  191         td->taskTick[i] = 0;
  192         td->clockTick[i] = 0.0;
  193         sampleHit[i] = 1;
  194       } else {
  195         td->taskTick[i] = (int_T)floor((td->period[i]-td->offset[i]) /
  196           stepSize+0.5);
  197         td->clockTick[i] = -1.0;
  198         sampleHit[i] = 0;
  199       }
  200     }
  201   
  202     /* Correct first sample time if continuous task */
  203     td->period[0] = stepSize;
  204     td->nTaskTicks[0] = 1;
  205   
  206     /* Set first discrete task index */
  207     if (rtmNumSampTimes == 1)
  208       td->firstDiscIdx = (int_T)(period[0] == CONTINUOUS_SAMPLE_TIME);
  209     else
  210       td->firstDiscIdx = ((int_T)(period[0] == CONTINUOUS_SAMPLE_TIME) +
  211                           (int_T)(period[1] == CONTINUOUS_SAMPLE_TIME));
  212     return(NULL);                        /* success */
  213   
  214   #endif                                 /* ! USE_RTMODEL case */
  215   
  216   }                                      /* end rt_SimInitTimingEngine */

  273   /* Function: rt_SimGetNextSampleHit ============================================
  274    * Abstract:
  275    *      For a single tasking real-time system, return time of next sample hit.
  276    */
  277   /* This function has a different signature in the RT_MALLOC case */
  278   #ifdef RT_MALLOC
  279   
  280   time_T rt_SimGetNextSampleHit(void *rtmTimingData,
  281     int_T rtmNumSampTimes)
  282   #else
  283     time_T rt_SimGetNextSampleHit(void)
  284   #endif
  285   {
  286   
  287   #ifdef USE_RTMODEL
  288   
  289     /* The USE_RTMODEL version of this function does nothing */
  290   #ifdef RT_MALLOC
  291   
  292     UNUSED_PARAMETER(rtmTimingData);
  293     UNUSED_PARAMETER(rtmNumSampTimes);
  294   
  295   #endif
  296   
  297     return -1;
  298   
  299   #else                                  /* must be !USE_RTMODEL */
  300   
  301     time_T timeOfNextHit;
  302   
  303   #ifdef RT_MALLOC
  304   
  305     TimingData *td;
  306     td = (TimingData *)rtmTimingData;
  307   
  308   #else
  309   
  310     TimingData *td;
  311     int_T rtmNumSampTimes;
  312     td = &td_struct;
  313     rtmNumSampTimes = NUMST;             /* it's not passed in, in this case */
  314   
  315   #endif
  316   
  317     td->clockTick[0] += 1;
  318     timeOfNextHit = td->clockTick[0] * td->period[0];
  319     if (rtmNumSampTimes > 1) {
  320       int i;
  321       for (i = 1; i < rtmNumSampTimes; i++) {
  322         if (++td->taskTick[i] == td->nTaskTicks[i]) {
  323           td->taskTick[i] = 0;
  324           td->clockTick[i]++;
  325         }
  326       }
  327     }
  328   
  329     return(timeOfNextHit);
  330   
  331   #endif                                 /* !USE_RTMODEL */
  332   
  333   }                                      /* end rt_SimGetNextSampleHit */

  335   /* Function: rt_SimUpdateDiscreteTaskSampleHits ================================
  336    * Abstract:
  337    *      This function is for use with single tasking real-time systems.
  338    *
  339    *      If the number of sample times is greater than one, then we need to
  340    *      update the discrete task sample hits for the next time step. Note,
  341    *      task 0 always has a hit since it's sample time is the fundamental
  342    *      sample time.
  343    */
  344   void rt_SimUpdateDiscreteTaskSampleHits(int_T rtmNumSampTimes,
  345     void *rtmTimingData,
  346     int_T *rtmSampleHitPtr,
  347     real_T *rtmTPtr)
  348   {
  349   
  350   #ifdef USE_RTMODEL
  351   
  352     /* The USE_RTMODEL version of this function does nothing */
  353     UNUSED_PARAMETER(rtmNumSampTimes);
  354     UNUSED_PARAMETER(rtmTimingData);
  355     UNUSED_PARAMETER(rtmSampleHitPtr);
  356     UNUSED_PARAMETER(rtmTPtr);
  357     return;
  358   
  359   #else                                  /* must be !USE_RTMODEL */
  360   
  361     int i;
  362     int_T *sampleHit;
  363   
  364   #ifdef RT_MALLOC
  365   
  366     TimingData *td;
  367     td = (TimingData *)rtmTimingData;
  368   
  369   #else
  370   
  371     TimingData *td;
  372     UNUSED_PARAMETER(rtmTimingData);
  373     td = &td_struct;
  374     rtmNumSampTimes = NUMST;
  375   
  376   #endif
  377   
  378     sampleHit = rtmSampleHitPtr;
  379     for (i = td->firstDiscIdx; i < rtmNumSampTimes; i++) {
  380       int_T hit = (td->taskTick[i] == 0);
  381       if (hit) {
  382         rttiSetTaskTime(rtmTPtr, i,
  383                         td->clockTick[i]*td->period[i] + td->offset[i]);
  384       }
  385   
  386       sampleHit[i] = hit;
  387     }
  388   
  389   #endif                                 /* !USE_RTMODEL */
  390   
  391   }                                      /* rt_SimUpdateDiscreteTaskSampleHits */
  392   
  393   #else                                  /* defined(MULTITASKING) */

  402   /* Function: rt_SimUpdateDiscreteEvents ========================================
  403    * Abstract:
  404    *      This function is for use with multitasking real-time systems.
  405    *
  406    *      This function updates the status of the RT_MODEL sampleHits
  407    *      flags and the perTaskSampleHits matrix which is used to determine
  408    *      when special sample hits occur.
  409    *
  410    *      The RT_MODEL contains a matrix, called perTaskSampleHits.
  411    *      This matrix is used by the ssIsSpecialSampleHit macro. The row and
  412    *      column indices are both task id's (equivalent to the root RT_MODEL
  413    *      sample time indices). This is a upper triangle matrix. This routine
  414    *      only updates the slower task hits (kept in column j) for row
  415    *      i if we have a sample hit in row i.
  416    *
  417    *                       column j
  418    *           tid   0   1   2   3   4   5
  419    *               -------------------------
  420    *             0 |   | X | X | X | X | X |
  421    *         r     -------------------------
  422    *         o   1 |   |   | X | X | X | X |      This matrix(i,j) answers:
  423    *         w     -------------------------      If we are in task i, does
  424    *             2 |   |   |   | X | X | X |      slower task j have a hit now?
  425    *         i     -------------------------
  426    *             3 |   |   |   |   | X | X |
  427    *               -------------------------
  428    *             4 |   |   |   |   |   | X |      X = 0 or 1
  429    *               -------------------------
  430    *             5 |   |   |   |   |   |   |
  431    *               -------------------------
  432    *
  433    *      How macros index this matrix:
  434    *
  435    *          ssSetSampleHitInTask(S, j, i, X)   => matrix(i,j) = X
  436    *
  437    *          ssIsSpecialSampleHit(S, my_sti, promoted_sti, tid) =>
  438    *              (tid_for(promoted_sti) == tid && !minor_time_step &&
  439    *               matrix(tid,tid_for(my_sti))
  440    *              )
  441    *
  442    *            my_sti       = My (the block's) original sample time index.
  443    *            promoted_sti = The block's promoted sample time index resulting
  444    *                           from a transition via a ZOH from a fast to a
  445    *                           slow block or a transition via a unit delay from
  446    *                           a slow to a fast block.
  447    *
  448    *      The perTaskSampleHits array, of dimension n*n, is accessed using
  449    *      perTaskSampleHits[j + i*n] where n is the total number of sample
  450    *      times, 0 <= i < n, and 0 <= j < n.  The C language stores arrays in
  451    *      row-major order, that is, row 0 followed by row 1, etc.
  452    *
  453    */
  454   time_T rt_SimUpdateDiscreteEvents(int_T rtmNumSampTimes,
  455     void *rtmTimingData,
  456     int_T *rtmSampleHitPtr,
  457     int_T *rtmPerTaskSampleHits)
  458   {
  459     int i, j;
  460     int_T *sampleHit;
  461   
  462   #ifdef RT_MALLOC
  463   
  464     TimingData *td;
  465     td = (TimingData *)rtmTimingData;
  466     rtmNumSampTimes = NUMST;
  467   
  468   #else
  469   
  470     TimingData *td;
  471     UNUSED_PARAMETER(rtmTimingData);
  472     td = &td_struct;
  473   
  474   #endif
  475   
  476     sampleHit = rtmSampleHitPtr;
  477   
  478     /*
  479      * Run this loop in reverse so that we do lower priority events first.
  480      */
  481     i = rtmNumSampTimes;
  482     while (--i >= 0) {
  483       if (td->taskTick[i] == 0) {
  484         /*
  485          * Got a sample hit, reset the counter, and update the clock
  486          * tick counter.
  487          */
  488         sampleHit[i] = 1;
  489         td->clockTick[i]++;
  490   
  491         /*
  492          * Record the state of all "slower" events
  493          */
  494         for (j = i + 1; j < rtmNumSampTimes; j++) {
  495           rttiSetSampleHitInTask(rtmPerTaskSampleHits, rtmNumSampTimes,
  496             j, i, sampleHit[j]);
  497         }
  498       } else {
  499         /*
  500          * no sample hit, increment the counter
  501          */
  502         sampleHit[i] = 0;
  503       }
  504   
  505       if (++td->taskTick[i] == td->nTaskTicks[i]) {/* update for next time */
  506         td->taskTick[i] = 0;
  507       }
  508     }
  509   
  510     return(td->clockTick[0]*td->period[0]);
  511   }                                      /* rt_SimUpdateDiscreteEvents */
  512   
  513   #endif                                 /* !USE_RTMODEL */

  515   /* Function: rt_SimUpdateDiscreteTaskTime ======================================
  516    * Abstract:
  517    *      This function is for use with multitasking systems.
  518    *
  519    *      After a discrete task output and update has been performed, this
  520    *      function must be called to update the discrete task time for next
  521    *      time around.
  522    */
  523   void rt_SimUpdateDiscreteTaskTime(real_T *rtmTPtr,
  524     void *rtmTimingData,
  525     int tid)
  526   {
  527   
  528   #ifdef USE_RTMODEL
  529   
  530     /* This function is a no-op in USE_RTMODEL */
  531     UNUSED_PARAMETER(rtmTPtr);
  532     UNUSED_PARAMETER(rtmTimingData);
  533     UNUSED_PARAMETER(tid);
  534     return;
  535   
  536   #else                                  /* must be !USE_RTMODEL */
  537   #ifdef RT_MALLOC
  538   
  539     TimingData *td;
  540     td = (TimingData *)rtmTimingData;
  541   
  542   #else
  543   
  544     TimingData *td;
  545     UNUSED_PARAMETER(rtmTimingData);
  546     td = &td_struct;
  547   
  548   #endif
  549   
  550     rttiSetTaskTime(rtmTPtr, tid,
  551                     td->clockTick[tid]*td->period[tid] + td->offset[tid]);
  552   
  553   #endif                                 /* !USE_RTMODEL */
  554   
  555   }
